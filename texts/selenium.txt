from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support.ui import Select
from selenium.webdriver.common.action_chains import ActionChains
from selenium.webdriver.common.alert import Alert
import time

# Initialize WebDriver
driver = webdriver.Chrome()

try:
    # Navigate to a demo page
    driver.get("https://www.example.com")

    # Example 1: Interacting with a text field
    text_field = driver.find_element(By.ID, "text_field_id")
    text_field.send_keys("Selenium Example")

    # Example 2: Clicking a button
    button = driver.find_element(By.ID, "button_id")
    button.click()

    # Example 3: Handling a checkbox
    checkbox = driver.find_element(By.ID, "checkbox_id")
    if not checkbox.is_selected():
        checkbox.click()

    # Example 4: Selecting a radio button
    radio_button = driver.find_element(By.ID, "radio_button_id")
    radio_button.click()

    # Example 5: Selecting a value from a dropdown
    dropdown = Select(driver.find_element(By.ID, "dropdown_id"))
    dropdown.select_by_visible_text("Option Text")  # By visible text
    dropdown.select_by_value("option_value")  # By value
    dropdown.select_by_index(1)  # By index

    # Example 6: Clicking a link
    link = driver.find_element(By.LINK_TEXT, "Example Link")
    link.click()

    # Example 7: Handling an alert
    driver.execute_script("alert('This is an alert!')")
    alert = Alert(driver)
    print(alert.text)  # Print alert text
    alert.accept()  # Accept alert

    # Example 8: Working with iframes
    driver.switch_to.frame("iframe_id")  # Switch to iframe
    iframe_element = driver.find_element(By.ID, "inside_iframe_element_id")
    iframe_element.click()
    driver.switch_to.default_content()  # Switch back to main content

    # Example 9: Interacting with a hoverable element
    hover_element = driver.find_element(By.ID, "hover_element_id")
    action = ActionChains(driver)
    action.move_to_element(hover_element).perform()

    # Example 10: Uploading a file
    file_input = driver.find_element(By.ID, "file_input_id")
    file_input.send_keys("C:/path/to/file.txt")

    # Example 11: Working with dynamic elements
    dynamic_element = driver.find_element(By.CSS_SELECTOR, "div.dynamic_class")
    print(dynamic_element.text)

    # Example 12: Fetching table data
    table = driver.find_element(By.ID, "table_id")
    rows = table.find_elements(By.TAG_NAME, "tr")
    for row in rows:
        cells = row.find_elements(By.TAG_NAME, "td")
        for cell in cells:
            print(cell.text)

    # Example 13: Scrolling to an element
    scroll_element = driver.find_element(By.ID, "scroll_element_id")
    driver.execute_script("arguments[0].scrollIntoView(true);", scroll_element)

    # Example 14: Drag and drop
    source = driver.find_element(By.ID, "source_element_id")
    target = driver.find_element(By.ID, "target_element_id")
    action.drag_and_drop(source, target).perform()

    # Example 15: Taking a screenshot
    driver.save_screenshot("screenshot.png")

finally:
    # Quit the driver
    time.sleep(5)  # Optional: Wait to observe actions
    driver.quit()
